  \draw (6,0) to[R, l=$R$] (6,2);
  \node at (6,-0.5) {Resistor};

  \node at (9,-0.5) {Capacitor};

  % Inductor
  \draw (12,0) to[L, l=$L$] (12,2);
  \node at (12,-0.5) {Inductor};
